# Firestore security rules for per-user cards
# Apply these in the Firebase console (Firestore Rules)

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Each user has their own subcollection 'cards' under users/{uid}/cards/{cardId}
    match /users/{userId}/cards/{cardId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Optionally, restrict top-level user documents
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
